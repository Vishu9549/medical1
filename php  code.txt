function calcDelivery(km){ return km > 10 ? null : Math.round(km * 8); }

function set(updates){ Object.assign(S, updates); render(); }

function cartCount(){ return Object.values(S.homeCart).reduce((a,b)=>a+b,0); }

function computeMatches(){
  const items = Object.keys(S.homeCart).map(id => CATALOG.find(m=>m.id===id)).filter(Boolean);
  if(!items.length) return [];
  return ALL_SHOPS.map(shop => {
    const available=[], missing=[];
    let tp=0;
    items.forEach(item => {
      if(item.inv && shop.inventory[item.inv]!==undefined){
        available.push({...item, shopPrice:shop.inventory[item.inv]});
        tp += shop.inventory[item.inv];
      } else { missing.push(item); }
    });
    const dc = calcDelivery(shop.distanceKm)||0;
    return {shop,available,missing,matchCount:available.length,totalPrice:tp,
            totalWithDelivery:tp+dc,deliveryCharge:dc,totalItems:items.length};
  }).sort((a,b)=>b.matchCount-a.matchCount||a.totalWithDelivery-b.totalWithDelivery);
}

function esc(s){ 
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

// ─── DOM HELPERS ──────────────────────────────────────────────
function el(tag, attrs={}, ...children){
  const e = document.createElement(tag);
  for(const [k,v] of Object.entries(attrs)){
    if(k==="cls") e.className=v;
    else if(k==="style" && typeof v==="object") Object.assign(e.style,v);
    else if(k.startsWith("on")) e.addEventListener(k.slice(2).toLowerCase(),v);
    else e.setAttribute(k,v);
  }
  children.flat().forEach(c=>{
    if(c==null||c===false||c===undefined) return;
    e.appendChild(typeof c==="string"||typeof c==="number" ? document.createTextNode(String(c)) : c);
  });
  return e;
}

function render(){
  const app = document.getElementById("app");
  app.innerHTML = "";
  const container = d({position:"relative"});
  let screenEl;
  switch(S.screen){
    case "home":        screenEl = renderHome(); break;
    case "profile":     screenEl = renderProfile(); break;
    case "smartcart":   screenEl = renderSmartCart(); break;
    case "cartResults": screenEl = renderCartResults(); break;
    case "orderSuccess":screenEl = renderOrderSuccess(); break;
    case "dashboard":   screenEl = renderDashboard(); break;
    case "admin":       screenEl = renderAdmin(); break;
    default:            screenEl = renderHome();
  }
  container.appendChild(screenEl);
  const modal = renderModal();
  if(modal) container.appendChild(modal);
  app.appendChild(container);
}
